Rene funksjoner


function pureFunction (input) {
  // no side-effect
  return input.split('').reverse().join('');
}
pureFunction()

Urene funksjoner


var foo = 2;
function unpureFunction (input) {
  // side-effect
  foo += 100;
  return input.split('').reverse().join('');
}

> foo //=> 2
> unpureFunction('foo') //=> oof
> foo //=> ????

Overraskende urene funksjoner


var obj = { foo: 'bar' };
function unpureFunction (input) {
  input.foo = 'foo';
  return input;
}

> obj.foo //=> 'bar'
> unpureFunction('foo').foo //=> 'foo'
> obj.foo //=> 'foo' <-- OOPS

function compose (...fns) {
  return (...args) =>
    fns.reduceRight((child, fn) =>
      fn.apply(this,
        child ? args.concat(child) : args),
      null);
};
